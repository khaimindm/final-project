create table Persons (
id int NOT NULL AUTO_INCREMENT,
login varchar(100) NOT NULL,
password varchar NOT NULL,
first_name varchar(255) NOT NULL,
last_name varchar(255) NOT NULL, 
birthday date NOT NULL,
role varchar(20),
PRIMARY KEY (id)
);
